         function plusQty(qtyInput){
            require([
                'jquery'
            ], function(jQuery){
                (function($) {
                    if($('#'+qtyInput).attr('disabled')!='disabled'){

                        qty = parseInt(qty);
                        qty++;
                        $('#'+qtyInput).val(qty);
                    }
                })(jQuery);
            });

            require([
                'jquery'
            ], function(jQuery){
                (function($) {

                        qty = parseInt(qty);

                        if(qty>1){
                            qty--;
                            $('#'+qtyInput).val(qty);
                        }
                    }
                })(jQuery);
            });
